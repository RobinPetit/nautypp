488  inline bool is_planar() const {
489  const auto n{this->V()};
490  // see nauty::planarg.c::isplanar
491  DYNALLSTAT(t_ver_sparse_rep, V, V_sz);
492  DYNALLSTAT(t_adjl_sparse_rep, A, A_sz);
493  DYNALLOC1(t_ver_sparse_rep, V, V_sz, n, "is_planar");
494  DYNALLOC1(t_adjl_sparse_rep, A, A_sz, 2*E()+1, "is_planar");
495  t_dlcl **dfs_tree, **back_edges, **mult_edges;
496  t_ver_edge *embed_graph;
497  int edge_pos, v, w, c;
498  // Conversion to nauty::planarity adjl format
499  int k{0};
500  for(Vertex v{0}; v < n; ++v) {
501  if(degree(v) == 0) {
502  V[v].first_edge = NIL;
503  } else {
504  V[v].first_edge = k;
505  for(Vertex w : neighbours_of(v)) {
506  A[k].end_vertex = w;
507  A[k].next = k+1;
508  ++k;
509  // No need to handle loop
510  }
511  A[k-1].next = NIL;
512  }
513  }
514  // see nauty::planarity.c
515  bool ret{static_cast<bool>(sparseg_adjl_is_planar(
516  V, n, A, &c,
517  &dfs_tree, &back_edges, &mult_edges,
518  &embed_graph, &edge_pos, &v, &w
519  ))};
520  sparseg_dlcl_delete(dfs_tree, n);
521  sparseg_dlcl_delete(back_edges, n);
522  sparseg_dlcl_delete(mult_edges, n);
523  embedg_VES_delete(embed_graph, n);
524  DYNFREE(V, V_sz);
525  DYNFREE(A, A_sz);
526  return ret;
527  }

573  static Graph from_adjacency_matrix_upper(const std::vector<T> A, size_t V) {
574  if(V*V != A.size())
575  throw std::runtime_error("Wrong dimensions for adjacency matrix");
576  Graph ret(V);
