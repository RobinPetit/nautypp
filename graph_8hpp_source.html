493  inline bool is_planar() const {
494  const auto n{this->V()};
495  // see nauty::planarg.c::isplanar
496  DYNALLSTAT(t_ver_sparse_rep, V, V_sz);
497  DYNALLSTAT(t_adjl_sparse_rep, A, A_sz);
498  DYNALLOC1(t_ver_sparse_rep, V, V_sz, n, "is_planar");
499  DYNALLOC1(t_adjl_sparse_rep, A, A_sz, 2*E()+1, "is_planar");
500  t_dlcl **dfs_tree, **back_edges, **mult_edges;
501  t_ver_edge *embed_graph;
502  int edge_pos, v, w, c;
503  // Conversion to nauty::planarity adjl format
504  int k{0};
505  for(Vertex v{0}; v < n; ++v) {
506  if(degree(v) == 0) {
507  V[v].first_edge = NIL;
508  } else {
509  V[v].first_edge = k;
510  for(Vertex w : neighbours_of(v)) {
511  A[k].end_vertex = w;
512  A[k].next = k+1;
513  ++k;
514  // No need to handle loop
515  }
516  A[k-1].next = NIL;
517  }
518  }
519  // see nauty::planarity.c
520  bool ret{static_cast<bool>(sparseg_adjl_is_planar(
521  V, n, A, &c,
522  &dfs_tree, &back_edges, &mult_edges,
523  &embed_graph, &edge_pos, &v, &w
524  ))};
525  sparseg_dlcl_delete(dfs_tree, n);
526  sparseg_dlcl_delete(back_edges, n);
527  sparseg_dlcl_delete(mult_edges, n);
528  embedg_VES_delete(embed_graph, n);
529  DYNFREE(V, V_sz);
530  DYNFREE(A, A_sz);
531  return ret;
532  }
